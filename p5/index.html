          var canvasWidth = 640;
          var canvasHeight = 480;
          function clearScreen() {
            background(0);
          }
          function setup() { //called once
            createCanvas(canvasWidth, canvasHeight);
            fill(255); //not lines
            stroke(255); //lines and borders
            clearScreen();
          }
          function draw() { //called per frame
            var centerX = canvasWidth/2;
            var centerY = canvasHeight/2;
            var starX = random(canvasWidth);
            var starY = random(canvasHeight);
            line(centerX, centerY, starX, starY);
            ellipse(starX, starY, 4, 6);      
          }
