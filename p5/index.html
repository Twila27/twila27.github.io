          var canvasWidth = 640;
          var canvasHeight = 480;
          function clearScreen() {
            background(0);
          }
          function setup() { //called once
            createCanvas(canvasWidth, canvasHeight);
            fill(255); //not lines
            stroke(255); //lines and borders
            clearScreen();
          }
          function draw() { //called per frame
            var centerX = canvasWidth/2;
            var centerY = canvasHeight/2;
            var starX = random(canvasWidth);
            var starY = random(canvasHeight);
            line(centerX, centerY, starX, starY);
            ellipse(starX, starY, 4, 6);      
          }
        
        var bubbleY = 100;
        var canvasWidth = 640;
        var canvasHeight = 480;
        function clearScreen() {
          background(46,64,114);
        }
        function setup() { //called once
          createCanvas(canvasWidth, canvasHeight);
          fill(255); //not lines
          stroke(255); //lines and borders
          frameRate(30);
          clearScreen();
        }
        function draw() { //called per frame
          clearScreen();
          var numBubbles = 15;
          var centerX = canvasWidth/2;
          var centerY = canvasHeight/2;

          bubbleY = bubbleY - 1;
          if ( bubbleY < 0 )
            bubbleY = height;

          for (bubbleIndex = 0; bubbleIndex < numBubbles; bubbleIndex++)
          {
              var bubbleX = random(canvasWidth);
              var bubbleW = random(1, 15);
              ellipse(bubbleX, bubbleY + random(15), bubbleW, bubbleW);
          }
        }
